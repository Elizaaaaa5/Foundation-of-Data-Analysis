def loglikelihood(beta, x, y):
    n = len(x)
    result = 0
    for i in range(n):
        result += ((y[i]-1)*(x[i].dot(beta)) - np.log(1+(np.e)**(-x[i].dot(beta))))
    return result

def gradient(beta, x, y):
    result = []
    n = len(x)
    d = len(beta)
    g = []
    for k in range(d):
        temp = 0
        for i in range(n):
            temp += (y[i]-1 + (np.e**(-(x[i].dot(beta))))/(1+np.e**(-(x[i].dot(beta)))))*x[i][k]
        g.append(temp)
    return np.array(g)

def fast_loglikelihood(beta, x, y):
    xb = np.matmul(x,beta)
    result = np.sum(np.matmul(y-1, xb)) - np.sum(np.log(1+np.e**(-xb)))
    return result


def fast_gradient(beta, x, y):
    xb = np.matmul(x,beta)
    var1 = y-1 
    var2 = np.e**(-xb)/(1+np.e**(-xb))
    var = np.array(var1+var2)
    res = var.dot(x)
    return (res)
